  def self.skin
    skin_tone = ["albino",
      "pale",
      "medium",
      "tanned",
      "brown",
      "chocolate",
      "black",
      "blue",
      "gold",
      "green",
      "silver",
      "translucent",
      "orange"
    ]
    return skin_tone[rand(skin_tone.length)]
  end
