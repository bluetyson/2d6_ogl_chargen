  def self.hair
    tone = ["light", "medium", "full" ]
    body = ["straight", "wavey", "curly", "frizzed"]
    color = ["blond", 
        "auburn",
        "brown",
        "chestnut",
        "red",
        "orange",
        "green",
        "blue",
        "black",
        "white",
        "gold",
        "silver",
        "yellow",
        "gray"
      ]
    length  = ["close cropped",
        "short",
        "medium length",
        "shoulder length",
        "very long",
        "waist length"
      ]

    t = tone[rand(tone.length)]
    b = body[rand(body.length)].capitalize
    c = color[rand(color.length)]
    l = length[rand(length.length)]
  
    hair = "#{b} #{t} #{c} #{l}"
    return hair
  end
