  def test_muster_out_cash
    assert(@char["character"].stuff.has_key?("cash"))
    min_cash = 2000 * (@char["terms"] / 2)
    max_cash = 40000 * (@char["terms"] / 2)
    assert(@char["character"].stuff["cash"] >= min_cash) 
    assert(@char["character"].stuff["cash"] <= max_cash) 
  end
