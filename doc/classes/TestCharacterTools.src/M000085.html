  def test_modify_stat
    @character.upp        = "777777"
    options               = Hash.new(0)
    options['character']  = @character
    options['stat']       = "Int"
    options['level']      = 2
    CharacterTools.modify_stat(options)
    assert(@character.upp == "777977")
  end
