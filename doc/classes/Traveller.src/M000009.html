  def Traveller.modify_stat(upp, index, difference)
    return upp if index > 5
    stat = upp[index, 1].to_i(16)
    new_stat = stat + difference
    new_stat = 0 if new_stat < 0
    new_stat = 15 if new_stat > 15
    upp[index] = new_stat.to_s(16).upcase
    return upp
  end
