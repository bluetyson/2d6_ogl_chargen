  def test_upp
    @upp = Traveller.upp
    assert(@upp.length == 6)
    assert(/[:xdigit:]{6}/, @upp)
    assert_instance_of(String, @upp)
    (0..5).each do |num|
      assert(@upp[num].chr.to_i(16) <= 15)
      assert(@upp[num].chr.to_i(16) >= 2)
    end
  end
