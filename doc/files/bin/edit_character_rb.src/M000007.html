def add_element(character, key)
  puts "Add (a)rray, (h)ash, or (s)tring?"
  element_type = gets.chomp!
  case element_type
    when "a"
      character[key] = Array.new
    when "h"
      character[key] = Hash.new(0)
    when "s"
      character[key] = ""
    else
      verbose_exit("Not sure what element type that is.")
    end
end
